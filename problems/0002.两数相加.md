# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution: # 写法1
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        ans = 0
        num1 = l1
        num2 = l2
        count = 0
        while num1!=None or num2!=None:
            if num1 and num2: ans+=(num1.val+num2.val)*10**count
            elif num1: ans+=(num1.val)*10**count
            elif num2: ans+=(num2.val)*10**count
            count+=1
            if num1: num1 = num1.next
            if num2: num2 = num2.next
        ans2 = str(ans)
        ans2 = ''.join(reversed(ans2))
        result = ListNode(val = int(ans2[0]))
        result1 = result
        for i in range(1,len(ans2)):

            result1 = result2
        return result
