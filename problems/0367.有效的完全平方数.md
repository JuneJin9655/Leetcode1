'''Python3
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        # 想个办法稍微缩小范围即可
        low,high = 0,num
        while low<= high:
            # 就用简单的二分搜索
            mid = (low + high) // 2
            if mid*mid < num:
                if (mid+1)*(mid+1) > num:
                    return False #反向了,说明不是完全平方
                else:
                    low = mid + 1 #注意边界
            elif mid*mid > num:
                if (mid-1)*(mid-1) < num:
                    return False
                else:
                    high = mid - 1
            else:
                return True
'''
